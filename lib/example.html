Get Started

Simple to use.

Add the glass effect to any element in seconds. 

// Import the library
import LiquidGlass from './liquid-glass.js';

// Apply to an element
const glass = new LiquidGlass('#my-element', {
  draggable: true,
  surfaceType: 'convex_squircle',
  bezelWidth: 30,
  glassThickness: 150
});
<!-- Include the styles -->
<link rel="stylesheet" href="liquid-glass.css">

<!-- Include the UMD script -->
<script src="liquid-glass.umd.js"></script>

<script>
  var glass = new LiquidGlass('#my-element', {
    draggable: true
  });
</script>
<!-- Use data attributes for configuration -->
<div
  data-liquid-glass
  data-lg-surface="convex_squircle"
  data-lg-bezel="30"
  data-lg-draggable="true"
></div>

<script>
  // Auto-initialize all elements
  LiquidGlass.autoInit();
</script>
